Esempio validazione dinamica con istruzioni javascript

jPagevalid.form('myFormConObjectJavascript')
.addInput({input: 'firstName',type: 'required'})
.addInput({input: 'total',type: 'required,number'})
.addInput({input: 'username',type: 'required,email'})
.addInput({input: 'country',type: 'required,alphanumeric'})
.addInput({input: 'state',type: 'required,alphanumeric'})
.addInput({input: 'zip',type: 'required,interger,cap'})
.addInput({input: 'save-info',type:'required'})
.checkOnSubmit()
.addValidation('alphanumeric',function(v){return /[A-Za-z0-9_]+/.test(v)},'Inserisci un testo valido')

Esempio validazione dinamica con inclusione di oggetto javascript

var obj_form = {
form: 'formTest4', //id form
submit: true,
inputs: [
{
input: 'firstName4',
boxErr: 'boxerror4',//opzionale - id div che contiene il msg di errore
type: 'required'
},
{
input: 'total4',
type: 'required,number',
keyup: true,
keypress: true
},
{
input: 'email4',
type: 'required,email',
},
{
input: 'testomin',
type: 'required',
valid: function (value,message,input,search) {// opzionale funzione aggiuntiva
if (value.length >19) {
return true;
} else
input.message = "Inserisci min. 20 caratteri : " + value.length
return false;
},
keyup: true,
keypress: true
}]}

jPagevalid.get('formTest4').onChange(function(valid){
if(!valid)
{
// fai qualcosa...
}
}).form(obj_form).checkOnSubmit();

Validare dei campi solo se richiesto. Esempio con l'utilizzo requiredIf:'ID-ELEMENTO O FUNZIONE'

var obj_formCheck = {
form: 'formTestCheck',
submit: true,
inputs: [
{input: 'firstName8',
type: 'required'
},
{input: 'total8',
type: 'required,number',
keyup: true,
keypress: true
},
{input: 'email8',
type: 'required,email',
},
{input: 'testomin8',
type: 'required',
valid: function (value,message,input,search) {
if (value.length >19) {
return true;
} else
input.message = "Inserisci min. 20 caratteri : " + value.length
return false;
},
keyup: true,
keypress: true
},{input: 'campo1',
requiredIf:'checkmaster',
keyup: true,
keypress: true
},{input: 'campo2',
requiredIf:'checkmaster',
keyup: true,
keypress: true
},{input: 'check1',
requiredIf:'checkmaster'
},{input: 'check2',
requiredIf:'checkmaster'
}]}
jPagevalid.form(obj_formCheck).checkOnSubmit();
